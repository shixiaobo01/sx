# 字符集
```
apt update && \
apt install  -y locales  && \
localedef -v -c -i en_US -f UTF-8 en_US.UTF-8 && \
localedef -v -c -i zh_CN -f utf8 zh_CN.utf8   && \
localedef -v -c -i zh_CN -f gbk zh_CN.gbk     && \
localedef -v -c -i zh_CN -f gb2312 zh_CN.gb2312
```
